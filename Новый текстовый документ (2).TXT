#include <iostream>
#include <string>
#include <locale.h>

using namespace std;

int main() {
    setlocale(LC_ALL, "Russian");
    
    string password;
    string correctPassword = "12345";
    
    cout << "[+] Введите пароль: ";
    cin >> password;
    
    while (password != correctPassword) {
        cout << "[-] Неверный пароль!\n";
        cout << "[+] Введите пароль: ";
        cin >> password;
    }
    
    cout << "[+] Добро пожаловать!\n";
    
    return 0;

}
